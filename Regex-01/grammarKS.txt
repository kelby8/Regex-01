grammar grammarKS ;

SEMI : ';'  ;
WHILE : 'while'  ;
LP : '(' ;
RP : ')' ;
IF : 'if' ;
ELSE : 'else' ;
RETURN : 'return' ;
PRINT : 'print' ;
INPUT : 'input' ;
OPEN : 'open' ;
READ : 'read' ;
WRITE : 'write' ;
CMA : ',' ;
CLOSE : 'close' ;
LBR : '{' ;
RBR : '}' ;
NUM : [0-9]+ ;
OR : 'or' ;
AND : 'and' ;
NOT : 'not' ;
RELOP : '>=' | '<=' | '>' | '<' | '!=' | '==' ;
PLUS : '+' ;
MULOP : '*' | '/' | '%' ;
MINUS : '-' ;
WHITESPACE : [ \t\r\n]+ -> skip ;

program : braceblock ;
stmts : stmt stmts |  ;
stmt : func_call SEMI |cond | loop | return_stmt SEMI ;
loop : WHILE LP expr RP braceblock ;
cond : IF LP expr RP braceblock | IF LP expr RP braceblock ELSE braceblock ;
braceblock : LBR stmts RBR ;
expr : orexp ;
orexp : orexp OR andexp | andexp ;
andexp : andexp AND notexp | notexp ;
notexp : NOT notexp | rel ;
rel : sum RELOP sum | sum ;
sum : sum PLUS term | sum MINUS term | term ;
term : term MULOP neg | neg ;
neg : MINUS neg | factor ;
factor : NUM | LP expr RP | func_call ;
return_stmt : RETURN expr ;
func_call : builtin_func_call ;
builtin_func_call : PRINT LP expr RP | INPUT LP RP | OPEN LP expr RP | READ LP expr RP | WRITE LP expr CMA expr RP | CLOSE LP expr RP;
